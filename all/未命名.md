@startuml
!define RECTANGLE class

RECTANGLE "前言" as Intro {
  :Nginx安装报错；
  :未找到nginx包；
  :通过反复试错找到解决方案；
}

RECTANGLE "步骤" as Steps {
  
  RECTANGLE "查看环境" as CheckEnv {
    :cat /etc/redhat-release;
    :查看环境信息;
  }

  RECTANGLE "查看网络环境" as CheckNetwork {
    :ping www.baidu.com;
    :检查服务器是否连接外网;
  }

  RECTANGLE "添加yum源" as AddYumRepo {
    :sudo rpm -ivh http://nginx.org/packages/centos/7/noarch/RPMS/nginx-release-centos-7-0.el7.ngx.noarch.rpm;
    :查看yum源：sudo yum repolist;
  }

  RECTANGLE "安装 Nginx" as InstallNginx {
    :sudo yum install nginx;
  }

  RECTANGLE "配置 Nginx服务" as ConfigureNginx {
    :sudo systemctl enable nginx;
    :sudo systemctl start nginx;
    :sudo systemctl reload nginx;
    :ps -ef | grep 'nginx';
  }

  RECTANGLE "防火墙设置" as FirewallSettings {
    RECTANGLE "打开防火墙" as OpenFirewall {
      :sudo firewall-cmd --zone=public --permanent --add-service=http;
      :sudo firewall-cmd --reload;
    }

    RECTANGLE "关闭防火墙" as CloseFirewall {
      :iptables -F;
    }
  }


}

Steps -down-> InstallNginx : 安装Nginx
Steps -down-> ConfigureNginx : 配置服务
Steps -down-> FirewallSettings : 设置防火墙
FirewallSettings -down-> OpenFirewall : 打开防火墙
FirewallSettings -down-> CloseFirewall : 关闭防火墙
Steps -down-> CheckWebsite : 查看网页
